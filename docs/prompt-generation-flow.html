Prompt Generation Flow

Tracks the Prompt panel from request body to rendered UI. Skill-map drills are LLM-only; endpoint fallback here means transport fallback, not synthetic drill creation.

flowchart TD
  subgraph REQUEST["A. Frontend deck request"]
    A1["requestedSkillMap"]
    A2["requestedTemplateMode"]
    A3["requestedTemplateTargets per patternSlug
pseudo + invariant scaffolds"]
    A4["fetchSkillMapDeck()"]
    A5["POST /api/coach/skill-map-drills-stream
fallback transport only: /api/coach/skill-map-drills"]
  end

  subgraph CONTEXT["B. Generator request context"]
    B1["build_generator_context()"]
    B2["system prompt requires each drill to include
prompt + templatePrompts + templateTargets"]
    B3["llm_payload carries
templateMode + templateTargets + skillMap + practiceHistory"]
  end

  subgraph ASSEMBLY["C. Drill assembly in generator.py"]
    C1["LLM drills only"]
    C2["_process_raw_drill(raw, index, body, generation_skill_map)"]

    C3["_template_targets_for_drill()"]
    C31["Start with body.templateTargets[patternSlug]
from frontend request"]
    C32["Overlay raw.templateTargets from the LLM if present"]
    C33["Always set algorithm target =
solution.replace('{{missing}}', missing)"]

    C4["_template_prompt_map()"]
    C41["If raw.templatePrompts[mode] exists:
use _clean_concise_prompt(...)"]
    C42["Else if target exists:
use _template_prompt_from_target(...)"]
    C43["Entry point comes from first non-empty line:
def solve(nums): / define bfs(...):"]

    C5["selected_prompt becomes card.prompt"]
    C51["templatePrompts[requestedTemplateMode]"]
    C52["or cleaned raw.prompt"]
    C53["or generated fallback prompt from full algorithm target"]
  end

  subgraph DELIVERY["D. Delivery back to App.tsx"]
    D1["stream 'drill' events or JSON response"]
    D2["setSkillMapDeck(drills)"]
    D3["card carries
prompt + templatePrompts + templateTargets"]
  end

  subgraph UI["E. Prompt selection in App.tsx"]
    E1["currentTemplateMode
pseudo | invariant | algorithm"]

    E2["practiceTarget"]
    E21["card.templateTargets[currentTemplateMode]"]
    E22["Else client fallback target:
buildPseudoTemplate() / buildInvariantTemplate() / full solution"]

    E3["generatedPracticePrompt"]
    E31["card.templatePrompts[currentTemplateMode]"]
    E32["or card.prompt"]

    E4["practicePrompt"]
    E41["generatedPracticePrompt"]
    E42["or buildPracticePrompt(currentTemplateMode, practiceTarget)"]

    E5["Rendered UI
Prompt

{practicePrompt}
"]
  end

  subgraph ADAPTIVE["F. Adaptive variation side path"]
    F1["/api/coach/adaptive-variation"]
    F2["returns templatePrompts = {templateMode: prompt}
templateTargets = {templateMode: specimen, algorithm: specimen}"]
    F3["feeds the same App.tsx prompt-selection path"]
  end

  subgraph FAILURE["G. Failure path"]
    G1["No provider or invalid/empty LLM output"]
    G2["Structured 503 error"]
    G3["Frontend shows generation unavailable state"]
  end
